#+begin_src sh
pactl get-default-sink alsa_output.pci-0000_00_1b.0.iec958-stereo
#+end_src

#+begin_src
pactl list short sinks 49 alsa_output.pci-0000_01_00.1.hdmi-stereo
#+end_src
PipeWire s32le 2ch 48000Hz SUSPENDED 50
alsa_output.pci-0000_00_1b.0.iec958-stereo PipeWire s32le 2ch 48000Hz SUSPENDED

#+begin_src sh
pactl set-default-sink 49 ❯ pactl get-default-sink

#+end_src
