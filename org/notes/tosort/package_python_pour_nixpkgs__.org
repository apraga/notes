#+title: Package Python pour nixpkgs
#+filetags: nix nixpkgs python

* Modèle
dans python-modules/MYPACKAGE/default.nix
#+begin_src nix
{
  config,
  lib,
  pkgs,
  typeguard,
  buildPythonPackage,
  packaging,
  pythonAtLeast,
  fetchFromGitHub,
  setuptools,
  wheel,
}:

buildPythonPackage rec {
  pname = "tensorflow-addons";
  version = "0.23.0";

  disabled = pythonAtLeast "3.12";

  src = fetchFromGitHub {
    owner = "tensorflow";
    repo = "addons";
    rev = "v${version}";
    sha256 = "sha256-2tIZsbB33JlSlvJ2QcE1s6l+G0NYt37V+nVII64qduQ=";
  };

  dependencies = [
    typeguard
    packaging
  ];

  pyproject = true;
  build-system = [
    setuptools
    wheel
  ];
}
#+end_src
